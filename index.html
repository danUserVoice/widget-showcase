/* ecmascript 6
* Include the UserVoice JavaScript SDK (only needed once on a page)
*/
UserVoice = window.UserVoice || [];
(() => {
  const uv = document.createElement('script');
  uv.type = 'text/javascript';
  uv.async = true;
  uv.src = 'https://widget.uservoice.com/{{ Javascript API key }}.js';
  let s = document.getElementsByTagName('script')[0];
  s.parentNode.insertBefore(uv,s);
})();

// Global Settings
UserVoice.push(['set', {
  accent_color: '#d74119', // Widget accent color
  trigger_color: '#FFF',   // Trigger symbol color
  trigger_background_color: '#d74119', // Trigger Background color
}]);

// Add default trigger to the bottom-right corner of the window:
UserVoice.push(['addTrigger', {
  mode: 'post_idea', // Other values: satisfaction, smartvote
  trigger_position: 'bottom-right'
}]);
